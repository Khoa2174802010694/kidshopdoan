<script>
    // Dữ liệu mẫu
    let cart = JSON.parse(localStorage.getItem("cart")) || [
        {id: 1, name: "Váy lụa hồng Pastel", price: 450000, qty: 1, image: "https://images.unsplash.com/photo-1515377905703-c4788e51af15?q=80&w=400&auto=format&fit=crop"},
        {id: 2, name: "Túi xách ngọc trai", price: 890000, qty: 1, image: "https://images.unsplash.com/photo-1584917865442-de89df76afd3?q=80&w=400&auto=format&fit=crop"}
    ];

    function render() {
        const cartBody = document.getElementById("cartBody");
        const cartCount = document.getElementById("cartCount");

        if (cart.length === 0) {
            cartBody.innerHTML = `
                <div class="empty-cart-ui">
                    <i class="fa-solid fa-bag-shopping fa-4x mb-3" style="color: #ffe5ec;"></i>
                    <h5 class="text-muted">Giỏ hàng đang chờ bạn đấy!</h5>
                    <a href="index" class="btn btn-sm mt-3 px-4" style="background: var(--primary-pink); color: white;">Mua sắm ngay</a>
                </div>`;
            updateTotals(0);
            cartCount.innerText = "0 sản phẩm";
            return;
        }

        cartBody.innerHTML = cart.map(item => `
            <div class="cart-item d-flex align-items-center">
                <img src="${item.image}" class="product-img me-3" alt="${item.name}">
                <div class="flex-grow-1">
                    <h6 class="fw-bold mb-1" style="font-size: 0.95rem;">${item.name}</h6>
                    <small class="text-muted">${item.price.toLocaleString()}₫</small>
                </div>
                <div class="d-flex align-items-center mx-3">
                    <button class="qty-btn" onclick="change(${item.id},-1)"><i class="fa-solid fa-minus"></i></button>
                    <span class="mx-3 fw-bold">${item.qty}</span>
                    <button class="qty-btn" onclick="change(${item.id},1)"><i class="fa-solid fa-plus"></i></button>
                </div>
                <div class="text-end" style="min-width: 110px;">
                    <span class="fw-bold" style="color: var(--deep-pink);">${(item.price * item.qty).toLocaleString()}₫</span>
                </div>
                <div class="btn-remove" onclick="removeItem(${item.id})">
                    <i class="fa-solid fa-trash-can"></i>
                </div>
            </div>
        `).join("");

        const total = cart.reduce((a, b) => a + b.price * b.qty, 0);
        updateTotals(total);
        cartCount.innerText = `${cart.length} sản phẩm`;
    }

    function updateTotals(total) {
        document.getElementById("subtotal").innerText = total.toLocaleString() + "₫";
        document.getElementById("total").innerText = total.toLocaleString() + "₫";
    }

    function change(id, d) {
        const item = cart.find(i => i.id == id);
        if (item) {
            item.qty += d;
            if (item.qty <= 0) return removeItem(id);
            save();
        }
    }

    function removeItem(id) {
        if(confirm("Quý khách muốn bỏ sản phẩm này khỏi giỏ hàng ạ?")) {
            cart = cart.filter(i => i.id != id);
            save();
        }
    }

    function save() {
        localStorage.setItem("cart", JSON.stringify(cart));
        render();
    }

    render();
</script>
