    // ===== ADMIN: DASHBOARD =====
    @GetMapping("/admin/dashboard")
    public String adminDashboard(Model model) {
        double totalRevenue = listOrder.stream()
                .filter(o -> "APPROVED".equals(o.getStatus()))
                .mapToDouble(Order::getTotal)
                .sum();

        model.addAttribute("totalProducts", listPro.size());
        model.addAttribute("totalOrders", listOrder.size());
        model.addAttribute("revenue", totalRevenue);

        List<Order> recent = new ArrayList<>(listOrder);
        Collections.reverse(recent);
        model.addAttribute("recentOrders", recent.stream().limit(5).collect(Collectors.toList()));

        return "admin/dashboard";
    }

    // ===== ADMIN: ORDER MANAGEMENT =====
    @GetMapping("/admin/orders")
    public String adminOrders(Model model) {
        List<Order> allOrders = new ArrayList<>(listOrder);
        Collections.reverse(allOrders);
        model.addAttribute("orders", allOrders);
        return "admin/orders";
    }

    @GetMapping("/admin/orders/update")
    public String updateOrderStatus(@RequestParam("id") String id, @RequestParam("status") String status) {
        listOrder.stream()
                .filter(o -> o.getId().equals(id))
                .findFirst()
                .ifPresent(o -> o.setStatus(status));
        return "redirect:/admin/orders";
    }

    // ===== ADMIN: CATEGORY =====
    @GetMapping("/admin/category")
    public String adminCategory(Model model) {
        model.addAttribute("category", new Category());
        model.addAttribute("categories", listCat);
        return "admin/category";
    }

    @PostMapping("/admin/category/save")
    public String saveCategory(@ModelAttribute("category") Category cat) {
        if (cat.getId() == null || cat.getId().isEmpty()) {
            cat.setId("CAT-" + UUID.randomUUID().toString().substring(0, 4).toUpperCase());
            listCat.add(cat);
        } else {
            listCat.removeIf(c -> c.getId().equals(cat.getId()));
            listCat.add(cat);
        }
        return "redirect:/admin/category";
    }

    @GetMapping("/admin/category/edit/{id}")
    public String editCategory(@PathVariable String id, Model model) {
        Category found = listCat.stream()
                .filter(c -> c.getId().equals(id))
                .findFirst()
                .orElse(new Category());
        model.addAttribute("category", found);
        model.addAttribute("categories", listCat);
        return "admin/category";
    }

    @GetMapping("/admin/category/delete/{id}")
    public String deleteCategory(@PathVariable String id) {
        listCat.removeIf(c -> c.getId().equals(id));
        return "redirect:/admin/category";
    }

    // ===== ADMIN: PRODUCT =====
    @GetMapping("/admin/product")
    public String adminProduct(Model model) {
        model.addAttribute("product", new Product());
        model.addAttribute("products", listPro);
        model.addAttribute("categories", listCat);
        return "admin/product";
    }

    @PostMapping("/admin/products/save")
    public String saveProduct(@ModelAttribute("product") Product p) {
        if (p.getId() == null || p.getId().isEmpty()) {
            p.setId("P" + (listPro.size() + 1));
            listPro.add(p);
        } else {
            listPro.removeIf(prod -> prod.getId().equals(p.getId()));
            listPro.add(p);
        }
        return "redirect:/admin/product";
    }

    @GetMapping("/admin/products/edit/{id}")
    public String editProduct(@PathVariable String id, Model model) {
        Product found = listPro.stream()
                .filter(p -> p.getId().equals(id))
                .findFirst()
                .orElse(new Product());
        model.addAttribute("product", found);
        model.addAttribute("products", listPro);
        model.addAttribute("categories", listCat);
        return "admin/product";
    }

    @GetMapping("/admin/products/delete/{id}")
    public String deleteProduct(@PathVariable String id) {
        listPro.removeIf(p -> p.getId().equals(id));
        return "redirect:/admin/product";
    }

    @GetMapping("/admin/coupon")

    @PostMapping("/admin/coupon/save")
    public String saveCoupon(@ModelAttribute("coupon") Coupon c) {
        if (c.getId() == null || c.getId().isEmpty()) {
            c.setId("C" + (listCoupon.size() + 1));
            listCoupon.add(c);
        } else {
            listCoupon.removeIf(x -> x.getId().equals(c.getId()));
            listCoupon.add(c);
        }
        return "redirect:/admin/coupon";
    }

    @GetMapping("/admin/coupon/edit/{id}")
    public String editCoupon(@PathVariable String id, Model model) {
        Coupon found = listCoupon.stream()
                .filter(c -> c.getId().equals(id))
                .findFirst()
                .orElse(new Coupon());
        model.addAttribute("coupon", found);
        model.addAttribute("coupons", listCoupon);
        return "admin/coupon";
    }

    @GetMapping("/admin/coupon/delete/{id}")
    public String deleteCoupon(@PathVariable String id) {
        listCoupon.removeIf(c -> c.getId().equals(id));
        return "redirect:/admin/coupon";

    @GetMapping("/admin/customer/delete/{id}")
    public String deleteUser(@PathVariable String id) {
        listUser.removeIf(u -> u.get("id").equals(id));
        return "redirect:/admin/customer";
